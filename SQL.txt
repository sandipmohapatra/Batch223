DataBase
-------------------
It is use to store data in the grid format.
Grid means rows and columns.
example :- we have a student table.
student table has rows and columns 
rollno	name	address 	phoneno
-------------------------------------------------------------
1	sandip	bangalore	7766554433
2	kiran	bangalore	7788666555
------------------------------------------------------------------
rollno is a column.name is a column,address is a column,phoneno is a column.
-----------------------------------------------------------------------
each column has a datatype
example:- rollno number,name is varchar2(30),address varchar2(30),phoneno number;
-------------------------------
how to create a student table?
create table student223(rollno number,name varchar2(30),address varchar2(30),phoneno number);
-------------------------------------------
to insert data?

--------------------------------------------------------------------------------------------
1)arithmatic operators(+,-,*,/)
----------------------------------
select rollno,name,address,salary+1000 from student223;

select rollno,name,address,salary+1000 as Incrementalsalary from student223;
------------------------------------------
select rollno,name,address,salary-1000 from student223;

select rollno,name,address,salary-1000 as Decrementalsalary from student223;
---------------------------------------------
select rollno,name,address,salary*12 from student223;

select rollno,name,address,salary*12 as Anualsalary from student223;
---------------------------------------------------------------
select rollno,name,address,salary/30 from student223;

select rollno,name,address,salary/30 as PerDaysalary from student223;
------------------------------------------------------------------------------------------------------------
2)comparison operator(=,<,>,<=,>=,!=(not equal to)
------------------------------------------------------------------------
select * from student223 where salary=76000;
select * from student223 where salary>=76000;
select * from student223 where salary<=76000;
select * from student223 where salary!=76000;
select * from student223 where salary<76000;
select * from student223 where salary>36000;
------------------------------------------------------------------------------------
3)comparison operator :- between and,not between,in,not in
-----------------------------------------------------------------------------------------
select * from student223 where salary between 35000 and 75000;

select * from student223 where salary not between 35000 and 75000;

select * from student223 where salary in(36000,76000);

select * from student223 where salary not in(34000,76000);
